1 Motivation

This file define XXX_LHS like classes that are useful to declar left hand side. By example: 

V = V + a * W

can be (por must be, if you care about performance) rewritten as: 

V = lhs(V) + a * W
Note
it would have been better to be able to write something like
V = lhs + a * W

however that's not possible as we use V information like size and element_type to construct the Vector_LHS class

2 Implementation

2.1 The lhs() function   User

This is the function the user must use. It returns a The Vector_LHS or The Matrix_LHS.

Returns a lhs Vector object 

  template <typename IMPL>
  Vector_LHS<IMPL>
  lhs(const Vector_Crtp<IMPL>& vector)
  {
    return {vector.impl()};
  }

Returns a lhs Matrix object 

  template <typename IMPL>
  Matrix_LHS<IMPL>
  lhs(const Matrix_Crtp<IMPL>& matrix)
  {
    return {matrix.impl()};
  }

2.1.1 Usage example

A blas like expression

\begin{equation*} M=\alpha M_1 M_2 + \beta M \end{equation*}

must be written as: 

M = a * M1 * M2+ b * rhs(M)

2.2 The Vector_LHS   Dev

  template <typename VECTOR_TYPE>
  class Vector_LHS : public Vector_Crtp<Vector_LHS<VECTOR_TYPE>>
  {
    static_assert(Is_Crtp_Interface_Of_v<Vector_Crtp, VECTOR_TYPE>);

    // ////////////////
    // Types
    // ////////////////
    //
   public:
    using base_type   = Vector_Crtp<Vector_LHS<VECTOR_TYPE>>;
    using exact_type  = typename base_type::exact_type;
    using traits_type = typename base_type::traits_type;

    using wrapped_type = typename VECTOR_TYPE::exact_type;

    using element_type = typename traits_type::element_type;
    using size_type    = typename traits_type::size_type;

    // ////////////////
    // Members
    // ////////////////
    //
    const VECTOR_TYPE& _vector;

    // ////////////////
    // Constructors
    // ////////////////
    //
   public:
    Vector_LHS(const VECTOR_TYPE& vector) : _vector(vector) {}

    const wrapped_type&
    wrapped() const
    {
      return _vector;
    }

    // Cannot be copied
    Vector_LHS& operator=(const Vector_LHS& vector) = delete;
  };

2.3 The Matrix_LHS   Dev

  template <typename MATRIX_TYPE>
  class Matrix_LHS : public Matrix_Crtp<Matrix_LHS<MATRIX_TYPE>>
  {
    static_assert(Is_Crtp_Interface_Of_v<Matrix_Crtp, MATRIX_TYPE>);

    // ////////////////
    // Types
    // ////////////////
    //
   public:
    using base_type   = Matrix_Crtp<Matrix_LHS<MATRIX_TYPE>>;
    using exact_type  = typename base_type::exact_type;
    using traits_type = typename base_type::traits_type;

    using wrapped_type = typename MATRIX_TYPE::exact_type;
    
    using element_type = typename traits_type::element_type;
    using I_size_type  = typename traits_type::I_size_type;
    using J_size_type  = typename traits_type::J_size_type;

    // ////////////////
    // Members
    // ////////////////
    //
    const wrapped_type& _matrix;

    // ////////////////
    // Constructors
    // ////////////////
    //
   public:
    Matrix_LHS(const MATRIX_TYPE& matrix) : _matrix(matrix) {}

    const wrapped_type&
    wrapped() const
    {
      return _matrix;
    }

    // Cannot be copied
    Matrix_LHS& operator=(const Matrix_LHS& matrix) = delete;
  };
