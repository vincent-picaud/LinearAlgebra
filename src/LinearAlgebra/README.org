  'hello.hpp',
  # Dense vector/matrix user header 
  'vector.hpp',
  'matrix.hpp'

			    include_directories : inc,
			    install : true,
			    sources: [LinearAlgebra_headers,LinearAlgebra_sources])

				       link_with : [LinearAlgebra_lib,],
				       dependencies : [blas_lib,])
